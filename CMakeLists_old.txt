cmake_minimum_required (VERSION 2.6)

project(mgps)
set(CMAKE_BUILD_TYPE Release)

set(MY_DIR ${mgps_SOURCE_DIR})

message( STATUS ${CMAKE_SYSTEM_PROCESSOR} "   " ${CMAKE_SYSTEM})
if(${CMAKE_SYSTEM_PROCESSOR} MATCHES "aarch64" AND ${CMAKE_SYSTEM} MATCHES "Linux") 
  message( STATUS "Building for:  " ${CMAKE_SYSTEM} "(" ${CMAKE_SYSTEM_PROCESSOR} ")" )
  include(config_tegra.cmake)
  option(ON_TEGRA "Option description" ON)
  add_definitions(-DON_TEGRA)
elseif(${CMAKE_SYSTEM_PROCESSOR} MATCHES "x86_64" AND ${CMAKE_SYSTEM} MATCHES "Linux") 
  message( STATUS "Building for:  " ${CMAKE_SYSTEM} "(" ${CMAKE_SYSTEM_PROCESSOR} ")" )
  include(config_amd64.cmake)
  option(ON_AMD64 "Option description" ON)
  add_definitions(-DON_AMD64)
elseif(${CMAKE_SYSTEM} MATCHES "Darwin")
  message( STATUS "Building for:  " ${CMAKE_SYSTEM} "  FlyCapture not available")
  include(config_mac.cmake)
  option(ON_MAC "Option description" ON)
  add_definitions(-DON_MAC)
endif()


add_subdirectory(imgui_lib)
include_directories(imgui_lib)

add_subdirectory(src)
include_directories(src)

add_subdirectory(opengl3_gui)
